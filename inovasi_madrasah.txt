    <script>
        // 1. Mobile Menu Toggle
        const mobileMenuBtn = document.getElementById('mobile-menu');
        const navLinks = document.querySelector('.nav-links');

        mobileMenuBtn.addEventListener('click', () => {
            navLinks.classList.toggle('active');
            const icon = mobileMenuBtn.querySelector('i');
            if (navLinks.classList.contains('active')) {
                icon.classList.remove('fa-bars');
                icon.classList.add('fa-times');
            } else {
                icon.classList.remove('fa-times');
                icon.classList.add('fa-bars');
            }
        });

        // 2. Animated Numbers (Stats)
        const stats = document.querySelectorAll('.stat-number');
        let hasAnimatedStats = false;

        const animateStats = () => {
            stats.forEach(stat => {
                const target = +stat.getAttribute('data-target');
                const suffix = stat.innerText.replace(/[0-9]/g, ''); // Get suffix like % or /
                const increment = target / 50; // Speed control

                const updateCount = () => {
                    const count = +stat.innerText.replace(/[^0-9]/g, '');
                    if (count < target) {
                        stat.innerText = Math.ceil(count + increment) + suffix;
                        setTimeout(updateCount, 30);
                    } else {
                        stat.innerText = target + suffix;
                    }
                };
                updateCount();
            });
        };

        // Trigger animation when stats section is in view
        const statsSection = document.getElementById('stats');
        const observer = new IntersectionObserver((entries) => {
            if (entries[0].isIntersecting && !hasAnimatedStats) {
                animateStats();
                hasAnimatedStats = true;
            }
        });
        observer.observe(statsSection);

        // 3. Tracking Simulation Logic
        const trackingForm = document.getElementById('trackingForm');
        const resultArea = document.getElementById('resultArea');
        const ticketInput = document.getElementById('ticketId');

        trackingForm.addEventListener('submit', function(e) {
            e.preventDefault();
            
            // Simulasi Loading
            const btn = this.querySelector('button');
            const originalText = btn.innerText;
            btn.innerText = 'Mencari...';
            btn.disabled = true;

            setTimeout(() => {
                // Reset tombol
                btn.innerText = originalText;
                btn.disabled = false;

                // Tampilkan Result Area (Mockup Logic)
                // Dalam aplikasi nyata, ini akan fetch ke API
                const ticketVal = ticketInput.value;
                if(ticketVal.trim() === "") {
                    showToast("Error", "Mohon masukkan nomor tiket.", "error");
                    return;
                }

                // Update UI hasil dengan input user biar terasa real
                document.getElementById('resTicket').innerText = `Tiket #${ticketVal}`;
                resultArea.style.display = 'block';
                
                // Scroll ke hasil
                resultArea.scrollIntoView({ behavior: 'smooth', block: 'nearest' });

                showToast("Berhasil", "Data tiket ditemukan.", "success");

            }, 1500); // Delay simulasi 1.5 detik
        });

        // 4. Custom Toast Notification System (Pengganti Alert)
        function showToast(title, message, type) {
            const container = document.getElementById('toast-container');
            
            // Buat elemen toast
            const toast = document.createElement('div');
            toast.className = 'toast';
            
            // Tentukan ikon berdasarkan tipe
            let iconClass = 'fa-info-circle';
            let colorClass = 'var(--primary)';
            
            if (type === 'success') {
                iconClass = 'fa-check-circle';
                toast.style.borderLeftColor = '#4caf50';
            } else if (type === 'error') {
                iconClass = 'fa-exclamation-circle';
                toast.style.borderLeftColor = '#f44336';
            }

            toast.innerHTML = `
                <i class="fas ${iconClass}" style="color: ${type === 'error' ? '#f44336' : 'var(--primary)'}"></i>
                <div class="toast-content">
                    <h4>${title}</h4>
                    <p>${message}</p>
                </div>
            `;

            container.appendChild(toast);

            // Trigger animasi masuk
            requestAnimationFrame(() => {
                toast.classList.add('show');
            });

            // Hapus setelah 3 detik
            setTimeout(() => {
                toast.classList.remove('show');
                setTimeout(() => {
                    toast.remove();
                }, 400); // Tunggu animasi keluar
            }, 3000);
        }

        // 5. Smooth Scroll untuk Anchor Links
        document.querySelectorAll('a[href^="#"]').forEach(anchor => {
            anchor.addEventListener('click', function (e) {
                e.preventDefault();
                navLinks.classList.remove('active'); // Tutup menu mobile jika terbuka
                
                // Update icon menu
                const icon = mobileMenuBtn.querySelector('i');
                icon.classList.remove('fa-times');
                icon.classList.add('fa-bars');

                document.querySelector(this.getAttribute('href')).scrollIntoView({
                    behavior: 'smooth'
                });
            });
        });
